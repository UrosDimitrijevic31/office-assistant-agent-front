# .claude/env-template.txt - office-assistant-web

# Backend API
NEXT_PUBLIC_API_URL=http://localhost:3001
NEXT_PUBLIC_API_TIMEOUT=30000

# Authentication
NEXT_PUBLIC_AUTH_COOKIE_NAME=auth-token
AUTH_REDIRECT_URL=/dashboard

# UI Configuration
NEXT_PUBLIC_DEFAULT_THEME=system
NEXT_PUBLIC_TOAST_DURATION=5000

# React Query
NEXT_PUBLIC_QUERY_RETRY_COUNT=3
NEXT_PUBLIC_QUERY_CACHE_TIME=5

# File Upload
NEXT_PUBLIC_MAX_FILE_SIZE=10485760
NEXT_PUBLIC_ALLOWED_FILE_TYPES=.pdf,.docx,.txt

# Feature Flags
NEXT_PUBLIC_ENABLE_DEVTOOLS=true
NEXT_PUBLIC_ENABLE_LOGGING=true

## Local Setup
cp .env.example .env.local
# Edit .env.local

npm install
npm run dev
# Runs on http://localhost:3000
